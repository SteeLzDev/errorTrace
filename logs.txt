SELECT table_name, table_rows
FROM information_schema.tables
WHERE table_schema = DATABASE()
ORDER BY table_rows DESC
LIMIT 50;


SELECT
  c.table_name,
  c.ordinal_position,
  c.column_name,
  c.column_type,
  c.is_nullable,
  c.column_key
FROM information_schema.columns c
WHERE c.table_schema = DATABASE()
ORDER BY c.table_name, c.ordinal_position;

SELECT table_name, COUNT(*) qtd_colunas
FROM information_schema.columns
WHERE table_schema = DATABASE()
GROUP BY table_name
ORDER BY qtd_colunas DESC;


SELECT table_name
FROM information_schema.columns
WHERE table_schema = DATABASE()
  AND column_name IN ('ser_codigo','SER_CODIGO','cse_codigo','CSE_CODIGO')
GROUP BY table_name
HAVING SUM(column_name IN ('ser_codigo','SER_CODIGO')) > 0
   AND SUM(column_name IN ('cse_codigo','CSE_CODIGO')) > 0
ORDER BY table_name;

SELECT ser_codigo, cse_codigo, COUNT(*) qtd
FROM NOME_DA_TABELA
GROUP BY ser_codigo, cse_codigo
ORDER BY qtd DESC
LIMIT 20;


SELECT *
FROM NOME_DA_TABELA
WHERE TRIM(ser_codigo) = '111111'
   OR TRIM(SER_CODIGO) = '111111'
LIMIT 50;


SELECT CONCAT(
  'SELECT ''', table_name, ''' AS tabela, COUNT(*) qtd ',
  'FROM ', table_name, ' WHERE TRIM(ser_codigo) = ''111111'';'
) AS q
FROM information_schema.columns
WHERE table_schema = DATABASE()
  AND column_name = 'ser_codigo'
ORDER BY table_name;


SELECT CONCAT(
  'SELECT ''', table_name, ''' AS tabela, COUNT(*) qtd ',
  'FROM ', table_name, ' WHERE TRIM(usu_codigo) = ''111111'';'
) AS q
FROM information_schema.columns
WHERE table_schema = DATABASE()
  AND column_name = 'usu_codigo'
ORDER BY table_name;
