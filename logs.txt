SELECT us.usu_codigo, us.ser_codigo
FROM tb_usuario_ser us
WHERE us.usu_codigo = '111111';


SELECT ra.ser_codigo, ra.cse_codigo
FROM tb_rel_auditoria ra
WHERE ra.ser_codigo = '111111'
GROUP BY ra.ser_codigo, ra.cse_codigo;


SELECT ra.cse_codigo, COUNT(*) qtd
FROM tb_rel_auditoria ra
WHERE ra.ser_codigo = '111111'
GROUP BY ra.cse_codigo
ORDER BY qtd DESC;


SELECT r.ser_codigo, r.cse_codigo
FROM tb_rel_auditoria_matricula_especial r
WHERE r.ser_codigo = '111111'
GROUP BY r.ser_codigo, r.cse_codigo;


SELECT r.ser_codigo, r.cse_codigo
FROM tb_rel_aud_matricula_especial r
WHERE r.ser_codigo = '111111'
GROUP BY r.ser_codigo, r.cse_codigo;


SELECT cse_codigo, cse_nome
FROM tb_consignante
WHERE cse_codigo = '<CSE_ENCONTRADO>';
